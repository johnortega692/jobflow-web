/** Max brush-out color columns per job (sheet headers expand as needed). */
var BRUSHOUT_MAX_COLORS = 25;

function doGet(e) {
  var params = (e && e.parameter) ? e.parameter : {};

  // JobFlow job setup — PM / Super dropdowns (JSON response, not HTML)
  if (params.action === 'getStaffLists') {
    return getFieldRequestStaffLists_();
  }

  // Check if a specific view is requested
  var view = params.view || 'form'; // Default to form view
  
  if (view === 'dashboard') {
    return HtmlService.createHtmlOutputFromFile('Dashboard')
      .setTitle('Material Order Dashboard')
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
  } else if (view === 'brushout') {
    // NEW: Add brushout form as a web app view
    return HtmlService.createHtmlOutputFromFile('BrushoutForm')
      .setTitle('Brushout Colors Manager')
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
  } else {
    // Your existing form rendering code
    return HtmlService.createHtmlOutputFromFile('Form')
      .setTitle('Ironwood Commercial Builders - Field Request Order')
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
  }
}

function getSetting(key) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('Settings');
  if (!sheet) throw new Error('Settings sheet not found. Please create a Settings sheet.');
 
  var data = sheet.getRange('A2:B20').getValues();
  for (var i = 0; i < data.length; i++) {
    var rowKey = data[i][0] ? data[i][0].toString().trim() : '';
    if (rowKey === key) {
      return data[i][1] ? data[i][1].toString().trim() : '';
    }
  }
  throw new Error('Setting not found: ' + key);
}

function getFieldRequestStaffLists_() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  function namesFromSheet(sheetName) {
    var sheet = ss.getSheetByName(sheetName);
    if (!sheet || sheet.getLastRow() < 2) return [];
    var values = sheet.getRange(2, 1, sheet.getLastRow() - 1, 1).getValues();
    var out = [];
    values.forEach(function(row) {
      var name = (row[0] || '').toString().trim();
      if (name && out.indexOf(name) === -1) out.push(name);
    });
    return out.sort();
  }
  return ContentService.createTextOutput(JSON.stringify({
    success: true,
    pms: namesFromSheet('PMs'),
    supers: namesFromSheet('Supers')
  })).setMimeType(ContentService.MimeType.JSON);
}
function upsertJobInJobsSheet_(data) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('Jobs');

  if (!sheet) {
    sheet = ss.insertSheet('Jobs');
    sheet.getRange(1, 1, 1, 5).setValues([['Job Code', 'Job Name', 'PM', 'Super', 'Address']]);
    sheet.getRange(1, 1, 1, 5).setFontWeight('bold');
    sheet.setFrozenRows(1);
  }

  var jobCode = (data.jobCode || '').toString().trim();
  var jobName = (data.jobName || '').toString().trim();
  var pm = (data.pm || '').toString().trim();
  var superName = (data.super || '').toString().trim();
  var address = (data.address || '').toString().trim();

  if (!jobCode) {
    return ContentService.createTextOutput(JSON.stringify({
      success: false,
      error: 'jobCode is required'
    })).setMimeType(ContentService.MimeType.JSON);
  }

  var lastRow = sheet.getLastRow();
  var rowToUpdate = -1;

  if (lastRow > 1) {
    var rows = sheet.getRange(2, 1, lastRow - 1, 5).getValues();
    for (var i = 0; i < rows.length; i++) {
      if ((rows[i][0] || '').toString().trim() === jobCode) {
        rowToUpdate = i + 2;
        break;
      }
    }
  }

  var rowData = [jobCode, jobName, pm, superName, address];
  if (rowToUpdate > 0) {
    sheet.getRange(rowToUpdate, 1, 1, 5).setValues([rowData]);
  } else {
    sheet.appendRow(rowData);
  }

  return ContentService.createTextOutput(JSON.stringify({
    success: true,
    message: 'Job ' + jobCode + ' saved to Jobs sheet.',
    jobCode: jobCode
  })).setMimeType(ContentService.MimeType.JSON);
} 
/**
 * JobFlow POST endpoints:
 *   ?action=upsertJob  — add/update Jobs sheet row
 *   ?sheet=brushouts   — merge brush-out colors (JobFlow / desktop)
 */
function doPost(e) {
  try {
    if (!e || !e.postData || !e.postData.contents) {
      return ContentService
        .createTextOutput(JSON.stringify({
          success: false,
          error: 'POST body required.'
        }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    if (e.parameter && e.parameter.action === 'upsertJob') {
      var jobData = JSON.parse(e.postData.contents);
      return upsertJobInJobsSheet_(jobData);
    }

    if (e.parameter && e.parameter.sheet === 'brushouts') {
      var brushoutsData = JSON.parse(e.postData.contents);
      return handleBrushoutsData(brushoutsData);
    }

    return ContentService
      .createTextOutput(JSON.stringify({
        success: false,
        message: 'Unknown POST. Use ?action=upsertJob or ?sheet=brushouts'
      }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    Logger.log('Error in doPost: ' + error.toString());
    return ContentService
      .createTextOutput(JSON.stringify({
        success: false,
        error: 'Failed to process request: ' + error.toString()
      }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function processBrushoutImport(jobNumber, pastedData, paintVendor) {
  if (!paintVendor || paintVendor.trim() === '') {
    return {
      success: 0,
      errors: ['Please select a paint vendor before importing.']
    };
  }
  
  // Remove quotes and normalize line breaks
  var cleanedData = pastedData
    .replace(/"/g, '')  // Remove all quotes
    .replace(/\t+/g, '\n')  // Replace tabs with line breaks
    .replace(/\r\n/g, '\n')  // Normalize line endings
    .replace(/\r/g, '\n');
  
  var lines = cleanedData.split('\n').filter(function(line) {
    return line.trim() !== '';
  });
  
  var colors = [];
  
  // Parse the pasted data - look for Color followed by Label
  for (var i = 0; i < lines.length; i++) {
    var trimmedLine = lines[i].trim();
    
    // Extract colors
    if (trimmedLine.toLowerCase().indexOf('color:') !== -1) {
      var color = trimmedLine.split('Color:')[1].trim();
      
      if (color && color !== '') {
        // Look ahead for the label on the next line
        var label = '';
        if (i + 1 < lines.length) {
          var nextLine = lines[i + 1].trim();
          if (nextLine.toLowerCase().indexOf('label:') !== -1) {
            label = nextLine.split('Label:')[1].trim();
          }
        }
        
        // Combine label with color if label exists
        var fullColor = label ? label + ': ' + color : color;
        colors.push(fullColor);
      }
    }
  }
  
  if (colors.length === 0) {
    return {
      success: 0,
      errors: ['Could not find any colors in the pasted data. Make sure to include the "Color:" rows.']
    };
  }

  if (colors.length > BRUSHOUT_MAX_COLORS) {
    return {
      success: 0,
      errors: ['Too many colors (' + colors.length + '). Maximum is ' + BRUSHOUT_MAX_COLORS + ' per job.']
    };
  }

  try {
    var result = handleBrushoutsData({
      job_number: jobNumber,
      paint_vendor: paintVendor,
      paint_items: colors,
      mode: 'merge'
    });
    var parsed = JSON.parse(result.getContent());
    if (!parsed.success) {
      return {
        success: 0,
        errors: [parsed.error || 'Import failed.']
      };
    }
    return {
      success: 1,
      errors: [],
      jobNumber: jobNumber,
      paintVendor: paintVendor,
      colorsFound: colors.length
    };
  } catch (e) {
    return {
      success: 0,
      errors: ['Error saving data: ' + e.message]
    };
  }
}

function splitJobCodeAndName(fullJob) {
  var s = (fullJob || '').toString().trim()
    .replace(/'/g, '')
    .replace(/'/g, '')
    .replace(/`/g, '');
  if (!s) return { code: '', name: '' };
  var spaceIdx = s.indexOf(' ');
  if (spaceIdx === -1) {
    return { code: s, name: '' };
  }
  return {
    code: s.substring(0, spaceIdx).trim(),
    name: s.substring(spaceIdx + 1).trim()
  };
}

function isLegacyFourColumnJobRow(row) {
  var eEmpty = !row[4] || !row[4].toString().trim();
  var d = row[3] ? row[3].toString().trim() : '';
  var dLooksLikeAddress = d && (/\d/.test(d) || /suite|street|\bst\b|ave|blvd|\bdr\b|way|arroyo|clay/i.test(d));
  var b = row[1] ? row[1].toString().trim() : '';
  return !!(eEmpty && dLooksLikeAddress && b);
}

function normalizeJobsRow(row) {
  if (!row || !row[0] || !row[0].toString().trim()) {
    return ['', '', '', '', ''];
  }

  if (isLegacyFourColumnJobRow(row)) {
    var split = splitJobCodeAndName(row[0]);
    return [
      split.code,
      split.name,
      row[1] ? row[1].toString().trim() : '',
      row[2] ? row[2].toString().trim() : '',
      row[3] ? row[3].toString().trim() : ''
    ];
  }

  return [
    row[0] ? row[0].toString().trim() : '',
    row[1] ? row[1].toString().trim() : '',
    row[2] ? row[2].toString().trim() : '',
    row[3] ? row[3].toString().trim() : '',
    row[4] ? row[4].toString().trim() : ''
  ];
}

function findJobRowByCode(jobsData, jobCode) {
  var code = jobCode.toString().trim();
  for (var i = 0; i < jobsData.length; i++) {
    var normalized = normalizeJobsRow(jobsData[i]);
    var fullLegacy = jobsData[i][0] ? jobsData[i][0].toString().trim() : '';
    if (normalized[0] === code) return normalized;
    if (fullLegacy === code) return normalized;
    if (normalized[0] + ' - ' + normalized[1] === code) return normalized;
  }
  return null;
}

function getDropdownData() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
 
  function getSheetData(sheetName, range) {
    var sheet = ss.getSheetByName(sheetName);
    if (!sheet) {
      console.error("Sheet not found:", sheetName);
      return [];
    }
    return sheet.getRange(range).getValues();
  }
 
  // Jobs: A=JobCode, B=JobName, C=PM, D=Super, E=Address (auto-normalizes legacy 4-col layout)
  var jobsData = getSheetData("Jobs", "A2:E");

  var jobs = jobsData
    .map(normalizeJobsRow)
    .filter(function(row) {
      return row[0] && row[0] !== '';
    });
 
  // Vendors data
  var vendorsData = getSheetData("Vendors", "A2:E");
 
  // Organize vendors by category
  var vendorsByCategory = {};
  vendorsData.forEach(function(vendor) {
    var category = vendor[4];
    if (category && category.trim() !== "") {
      if (!vendorsByCategory[category]) vendorsByCategory[category] = [];
      vendorsByCategory[category].push(vendor);
    }
  });
 
  // Paint vendors only
  var paintVendors = vendorsData.filter(function(vendor) {
    return vendor[4] && vendor[4].toString().trim() === 'Paint';
  });
 
  // Sundries
  var sundriesData = getSheetData("Sundries", "A2:C");
  var sundryCategories = {};
  sundriesData.forEach(function(row) {
    var category = row[1];
    var item = row[0];
    if (category && category.trim() !== "" && item && item.trim() !== "") {
      if (!sundryCategories[category]) sundryCategories[category] = [];
      sundryCategories[category].push(item);
    }
  });
 
  // Wallcovering
  var wallcoveringData = getSheetData("WC", "A2:B");
  var wallcoveringCategories = {};
  wallcoveringData.forEach(function(row) {
    var item = row[0];
    var category = row[1];
    if (category && category.trim() !== "" && item && item.trim() !== "") {
      if (!wallcoveringCategories[category]) wallcoveringCategories[category] = [];
      wallcoveringCategories[category].push(item);
    }
  });
 
  var rentalEquipment = getRentalEquipment();
  var equipment = getEquipment();
 
  return {
    vendors: vendorsData,
    paintVendors: paintVendors,
    vendorsByCategory: vendorsByCategory,
    pms: getSheetData("PMs", "A2:B"),
    supers: getSheetData("Supers", "A2:B"),
    sundryCategories: sundryCategories,
    sheen: getSheetData("Products", "B2:B"),
    products: getSheetData("Products", "A2:A"),
    jobs: jobs,  // [jobCode, jobName, pm, super, address]
    rentalEquipment: rentalEquipment,
    equipment: equipment,
    wallcoveringCategories: wallcoveringCategories
  };
}


function parseRentalData(rentalString) {
  return rentalString.split('\n').filter(Boolean).map(function(item) {
    var match = item.match(/(\d+)x\s+(.*?)\s*\((.*?)\)/);
    if (match) return [match[1], match[2], match[3]];
    var simplifiedMatch = item.match(/(.*?)\s*\((.*?)\)/);
    return simplifiedMatch ? ['1', simplifiedMatch[1], simplifiedMatch[2]] : ['1', item, ''];
  });
}

function parseEquipmentData(equipmentString) {
  return equipmentString.split('\n').filter(Boolean).map(function(item) {
    var match = item.match(/(\d+)x\s*(.*)/);
    return match ? [match[1], match[2]] : ['', item];
  });
} 
function submitWallcoveringForm(data) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();

      sheet = ss.insertSheet("WallcoveringSubmissions");
      sheet.appendRow(["Job", "Job Name", "Name", "Notes", "Date", "Wallcovering Items", "PM", "Super", "PM Email", "Super Email"]);
    }
 
    var inputDate = new Date(data.date);
    var userTimezoneOffset = inputDate.getTimezoneOffset() * 60000;
    var adjustedDate = new Date(inputDate.getTime() + userTimezoneOffset);
    var formattedDate = formatDate(adjustedDate);
 
    var jobCode = data.job;
    var jobName = data.jobName || '';
 
    sheet.appendRow([jobCode, jobName, data.name, data.notes, formattedDate, data.wallcoveringItems,
      data.pm, data.super, data.pmEmail, data.superEmail]);
 
    // Pull settings
    var templateId    = getSetting('Material Order Template');
    var senderName    = getSetting('Sender Name');
    var warehouseEmail = getSetting('Default Warehouse Email') || getEmailByCategory("Warehouse");
    var saveToDrive   = getSetting('Save PDF To Drive').toLowerCase() === 'yes';
    var pdfFolderName = getSetting('PDF Folder Name');
 
    var templateFile = DriveApp.getFileById(templateId);
    var copy = templateFile.makeCopy(jobCode + ' - ' + jobName + ' - Wallcovering Order');
    var doc = DocumentApp.openById(copy.getId());
    var body = doc.getBody();
 
    body.replaceText('{{job}}',      jobCode);
    body.replaceText('{{job_name}}', jobName);
    body.replaceText('{{name}}',     data.name);
    body.replaceText('{{notes}}',    data.notes);
    body.replaceText('{{date}}',     formattedDate);
    body.replaceText('{{pm}}',       data.pm);
    body.replaceText('{{super}}',    data.super);
    body.replaceText('{{vendor}}',   'Wallcovering Order');
 
    setDocumentStyle(doc);
 
    var itemsPlaceholder = body.findText('{{items}}');
    if (itemsPlaceholder) itemsPlaceholder.getElement().getParent().removeFromParent();
 
    insertItemsTable(body, 'Wallcovering', parseWallcoveringData(data.wallcoveringItems), ['Qty', 'Item']);
 
    var now = new Date();
    var stampParagraph = body.appendParagraph('Request submitted on: ' + now.toLocaleDateString() + ' at ' + now.toLocaleTimeString());
    applyTextStyle(stampParagraph, { italic: true, fontSize: 9, foregroundColor: '#666666', fontFamily: 'Arial' });
    stampParagraph.setAlignment(DocumentApp.HorizontalAlignment.RIGHT);
 
    doc.saveAndClose();
 
    var pdf = DriveApp.getFileById(copy.getId()).getAs('application/pdf');
 
    var ccRecipients = [];
    if (data.pmEmail     && data.pmEmail.trim()     !== '') ccRecipients.push(data.pmEmail);
    if (data.superEmail  && data.superEmail.trim()  !== '') ccRecipients.push(data.superEmail);
    if (data.superEmail2 && data.superEmail2.trim() !== '') ccRecipients.push(data.superEmail2);
    if (data.foreman     && data.foreman.trim()     !== '') ccRecipients.push(data.foreman);
    var ccString = ccRecipients.join(',');
 
    var emailBody = buildWallcoveringEmail(data, jobCode, jobName, formattedDate);
 
    GmailApp.sendEmail(warehouseEmail, jobCode + ' - ' + jobName + ' - Wallcovering Order', '', {
      cc: ccString, attachments: [pdf], name: senderName, htmlBody: emailBody
    });
 
    if (saveToDrive) {
      var folders = DriveApp.getFoldersByName(pdfFolderName);
      var folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(pdfFolderName);
      DriveApp.getFileById(copy.getId()).moveTo(folder);
    } else {
      DriveApp.getFileById(copy.getId()).setTrashed(true);
    }
 
    return 'Wallcovering order submitted successfully!';

// Helper function for parsing wallcovering data
function parseWallcoveringData(wallcoveringString) {
  const rows = [];
  if (!wallcoveringString) return rows;
  
  const lines = wallcoveringString.split('\n').filter(Boolean);
  
  // Track packout data and regular items separately
  let packoutName = null;
  let packoutItems = [];
  let regularItems = [];
  let packoutQty = "1"; // Default quantity
  
  // First pass - identify packouts and items
  lines.forEach(function(line) {
    const match = line.match(/(\d+)x\s*(.*)/);
    
    if (match) {
      const qty = match[1];
      const itemText = match[2];
      
      // If it's an actual packout header (explicitly has PACKOUT:)
      if (itemText.includes('PACKOUT:')) {
        packoutName = itemText.split('PACKOUT:')[1].trim();
        packoutQty = qty;
      }
      // If it's a packout item (has "Part of")
      else if (itemText.includes('(Part of ')) {
        // If we don't have a packout name yet, extract it from item
        if (!packoutName) {
          const parts = itemText.match(/\(Part of (.*?)\)/);
          if (parts && parts[1]) {
            packoutName = parts[1].trim();
          }
        }
        
        // Add to packout items array
        const baseItem = itemText.split('(Part of ')[0].trim();
        packoutItems.push([qty, baseItem]);
      }
      // Regular item
      else {
        regularItems.push([qty, itemText]);
      }
    } else {
      // Special case for the mysterious triangle
      if (line === "▼") {
        // Skip this line - don't include the triangle in the results
      } else {
        // Any non-matching lines
        regularItems.push(['', line]);
      }
    }
  });
  
  // Build the final ordered list
  
  // 1. First add the packout header if we found packout items
  if (packoutName && packoutItems.length > 0) {
    // Add packout header
    rows.push([packoutQty, `PACKOUT: ${packoutName}`]);
    
    // Then add packout items with four dashes instead of bullets
    packoutItems.forEach(function(item) {
      rows.push([item[0], `---- ${item[1]}`]);
    });
  }
  
  // 2. Then add all regular items
  regularItems.forEach(function(item) {
    rows.push(item);
  });
  
  return rows;
}




// Function to upload an image to Drive and return the file ID
function uploadImage(base64Data, fileName) {
  try {
    // Remove the prefix from the base64 string if it exists
    var imageData = base64Data.split(',')[1];
    if (!imageData) {
      imageData = base64Data;
    }
    
    // Decode the base64 string
    var blob = Utilities.newBlob(Utilities.base64Decode(imageData), 'image/jpeg', fileName);
    
    // Create a folder if it doesn't exist
    var folderName = 'HaulOffImages';
    var folders = DriveApp.getFoldersByName(folderName);
    var folder;
    
    if (folders.hasNext()) {
      folder = folders.next();
    } else {
      folder = DriveApp.createFolder(folderName);
    }
    
    // Upload the file to the folder
    var file = folder.createFile(blob);
    return file.getId();
  } catch (error) {
    console.error('Error uploading image:', error);
    return null;
  }
}

// ============================================================
// UPDATED getNextPONumber()
// Uses job CODE only (column A) — not the full combined string
// ============================================================
 
function getNextPONumber(jobCode) {
  // Strip job name if accidentally passed in combined format
  // e.g. "24-P2033 - St. Francis" → "24-P2033"
  var cleanCode = jobCode.toString().trim().split(' ')[0];
 
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var poSheet = ss.getSheetByName('POs');
 
  if (!poSheet) {
    poSheet = ss.insertSheet('POs');
    poSheet.appendRow([
      'PO Number', 'Job Code', 'Job Name', 'Vendor',
      'Date Issued', 'PM', 'Super',
      'Order Type', 'Items Summary'
    ]);
    var headerRange = poSheet.getRange(1, 1, 1, 9);
    headerRange.setBackground('#2c3e50');
    headerRange.setFontColor('#ffffff');
    headerRange.setFontWeight('bold');
    poSheet.setFrozenRows(1);
  }
 
  var lastRow = poSheet.getLastRow();
  var highestSeq = 0;
 
  if (lastRow > 1) {
    // PO Number is col A, Job Code is col B
    var data = poSheet.getRange(2, 1, lastRow - 1, 2).getValues();
 
data.forEach(function(row) {
  // Skip completely empty rows
  if (!row || !row[0] || !row[1]) return;

  var poNum  = row[0].toString().trim();
  var poCode = row[1].toString().trim().split(' ')[0];

  if (!poNum || !poCode) return;

  if (poCode === cleanCode) {
    var parts = poNum.split('-');
    if (parts.length >= 2) {
      var seq = parseInt(parts[parts.length - 1], 10);
      if (!isNaN(seq) && seq > highestSeq) highestSeq = seq;
    }
  }
});
  }
 
  var nextSeq = (highestSeq + 1).toString().padStart(3, '0');
  return cleanCode + '-' + nextSeq;
}

  var ss = SpreadsheetApp.getActiveSpreadsheet();

function submitForm(data) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Submissions");
 
    // Date handling
    var inputDate = new Date(data.date);
    var userTimezoneOffset = inputDate.getTimezoneOffset() * 60000;
    var adjustedDate = new Date(inputDate.getTime() + userTimezoneOffset);
    var formattedDate = formatDate(adjustedDate);
 
    var jobCode = data.job;
    var jobName = data.jobName || '';
 
    // Generate PO number
    var poNumber = getNextPONumber(jobCode);
    Logger.log('Generated PO: ' + poNumber);
 
    // Log to Submissions sheet
    sheet.appendRow([
      jobCode, jobName, data.name, data.notes, data.vendor, data.pm, data.super,
      data.foreman, formattedDate, data.paint, data.sundries, data.additional, poNumber
    ]);
 
    // Pull settings
    var templateId   = getSetting('Material Order Template');
    var senderEmail  = getSetting('Sender Email');
    var senderName   = getSetting('Sender Name');
    var companyName  = getSetting('Company Name');
    var saveToDrive  = getSetting('Save PDF To Drive').toLowerCase() === 'yes';
    var pdfFolderName = getSetting('PDF Folder Name');
 
    // Build PDF document
    var templateFile = DriveApp.getFileById(templateId);
    var copy = templateFile.makeCopy(jobCode + ' - ' + jobName + ' - Material Order - ' + poNumber);
    var doc = DocumentApp.openById(copy.getId());
    var body = doc.getBody();
 
    // Replace placeholders in body
    body.replaceText('{{job}}',       jobCode);
    body.replaceText('{{job_name}}',  jobName);
    body.replaceText('{{po_number}}', poNumber);
    body.replaceText('{{name}}',      data.name   || '');
    body.replaceText('{{notes}}',     data.notes  || '');
    body.replaceText('{{vendor}}',    data.vendor || '');
    body.replaceText('{{pm}}',        data.pm     || '');
    body.replaceText('{{super}}',     data.super  || '');
    body.replaceText('{{date}}',      formattedDate);
 
    // Footer
    var footer = doc.getFooter();
    if (!footer) {
      footer = doc.addFooter();
      var footerPara = footer.appendParagraph('PO# ' + poNumber + '  |  ' + jobCode + ' - ' + jobName);
      footerPara.setAlignment(DocumentApp.HorizontalAlignment.LEFT);
      applyTextStyle(footerPara, { fontSize: 8, foregroundColor: '#666666', fontFamily: 'Arial' });
    } else {
      footer.replaceText('\\{\\{po_number\\}\\}', poNumber);
      footer.replaceText('\\{\\{job\\}\\}',       jobCode);
      footer.replaceText('\\{\\{job_name\\}\\}',  jobName);
    }
 
    setDocumentStyle(doc);
 
    // Remove items placeholder
    var itemsPlaceholder = body.findText('{{items}}');
    if (itemsPlaceholder) {
      itemsPlaceholder.getElement().getParent().removeFromParent();
    }
 
    // Insert tables
    insertItemsTable(body, 'Paint',    parsePaintData(data.paint),       ['Qty', 'Product', 'Sheen', 'Color']);
    body.appendParagraph('').setLineSpacing(1.5);
    insertItemsTable(body, 'Sundries', parseSundriesData(data.sundries), ['Qty', 'Product']);
    body.appendParagraph('').setLineSpacing(1.5);
    insertAdditionalItems(body, 'Additional Items:', data.additional.split('\n').filter(Boolean));
 
    // Timestamp
    var now = new Date();
    var stampParagraph = body.appendParagraph(
      'Request submitted on: ' + now.toLocaleDateString() + ' at ' + now.toLocaleTimeString()
    );
    applyTextStyle(stampParagraph, { italic: true, fontSize: 9, foregroundColor: '#666666', fontFamily: 'Arial' });
    stampParagraph.setAlignment(DocumentApp.HorizontalAlignment.RIGHT);
 
    doc.saveAndClose();
 
    var pdf = DriveApp.getFileById(copy.getId()).getAs('application/pdf');
 
    // Build CC list
    var to = data.vendorEmail;
    var ccRecipients = [];
    if (data.vendorEmail2  && data.vendorEmail2.trim()  !== '') ccRecipients.push(data.vendorEmail2);
    if (data.pmEmail       && data.pmEmail.trim()       !== '') ccRecipients.push(data.pmEmail);
    if (data.superEmail    && data.superEmail.trim()    !== '') ccRecipients.push(data.superEmail);
    if (data.superEmail2   && data.superEmail2.trim()   !== '') ccRecipients.push(data.superEmail2);
    if (data.foreman       && data.foreman.trim()       !== '') ccRecipients.push(data.foreman);
    var ccString = ccRecipients.join(',');
 
    // Build and send email
    var paintItems    = parsePaintData(data.paint);
    var sundriesItems = parseSundriesData(data.sundries);
    var emailBody     = buildMaterialOrderEmail(data, jobCode, jobName, poNumber, formattedDate, paintItems, sundriesItems);
 
    GmailApp.sendEmail(
      to,
      jobCode + ' - ' + jobName + ' - Material Order - PO# ' + poNumber,
      '',
      {
        cc: ccString,
        attachments: [pdf],
        name: senderName,
        htmlBody: emailBody
      }
    );
 
    // Save or trash the doc
    if (saveToDrive) {
      var folders = DriveApp.getFoldersByName(pdfFolderName);
      var folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(pdfFolderName);
      DriveApp.getFileById(copy.getId()).moveTo(folder);
    } else {
      DriveApp.getFileById(copy.getId()).setTrashed(true);
    }
 
    // Log PO
    logPO(poNumber, jobCode, jobName, data.vendor, data.pm, data.super, 'Paint/Material',
      data.paint.split('\n').filter(Boolean).join(' | '));
 
    return { success: true, poNumber: poNumber, message: 'Form submitted successfully! PO# ' + poNumber };
 
  } catch (error) {
    console.error('Error in submitForm:', error);
    return { success: false, poNumber: null, message: 'An error occurred: ' + error.toString() };
  }
}

// Update the formatDate function to handle the timezone-adjusted date
function formatDate(date) {
  var inputDate = new Date(date);
  var month = (inputDate.getMonth() + 1).toString().padStart(2, '0');
  var day = inputDate.getDate().toString().padStart(2, '0');
  var year = inputDate.getFullYear();
  return month + '/' + day + '/' + year;
}

  titleParagraph.setSpacingAfter(6);

  var table = body.appendTable();
  var headerRow = table.appendTableRow();
  headerRow.setMinimumHeight(0);  // Set minimum height for header row
  
  // Add '#' header for numbering column
  var numberCell = headerRow.appendTableCell().setText('Line');
  applyTextStyle(numberCell, {
    bold: false,
    fontSize: 10,
    foregroundColor: '#FFFFFF',
    fontFamily: 'Calibri'
  });
  numberCell.setBackgroundColor('#000000');
  numberCell.setPaddingTop(0).setPaddingBottom(0).setVerticalAlignment(DocumentApp.VerticalAlignment.CENTER);

  headers.forEach(function(header) {
    var cell = headerRow.appendTableCell().setText(header);
    applyTextStyle(cell, {
      bold: true,
      fontSize: 10,
      foregroundColor: '#FFFFFF',
      fontFamily: 'Calibri'
    });
    cell.setBackgroundColor('#000000');
    cell.setPaddingTop(0).setPaddingBottom(0).setVerticalAlignment(DocumentApp.VerticalAlignment.CENTER);
  });

  items.forEach(function(item, rowIndex) {
    var row = table.appendTableRow();
    row.setMinimumHeight(0);  // Set minimum height for each row
    
    // Add numbering cell with period
    var numberCell = row.appendTableCell().setText((rowIndex + 1) + '.');
    applyTextStyle(numberCell, {
      bold: false,
      fontSize: 10,
      foregroundColor: '#000000',
      fontFamily: 'Calibri'
    });
    numberCell.setPaddingTop(0)
      .setPaddingBottom(0)
      .setVerticalAlignment(DocumentApp.VerticalAlignment.CENTER);

    // Process each cell in the row
    item.forEach(function(cellData, cellIndex) {
      // Process different types of content
      var formattedText = cellData;
      var isPackoutHeader = false;
      var isPackoutItem = false;
      
      if (typeof cellData === 'string') {
        // Check for packout headers
        if (cellData.includes('PACKOUT:')) {
          isPackoutHeader = true;
        } 
        // Check for packout items (lines that start with dashes and bullets)
        else if (cellData.startsWith("---- •")) {
          isPackoutItem = true;
          // Add spaces at the beginning for indentation instead of using setIndentStart
          formattedText = "        " + cellData.replace("---- •", "•");
        }
      }
      
      // Create the cell with the possibly modified text
      var cell = row.appendTableCell().setText(formattedText);
      
      // Apply styling based on cell type
      if (isPackoutHeader) {
        applyTextStyle(cell, {
          bold: true,
          fontSize: 10,
          foregroundColor: '#2c3e50',  // Dark blue for packout headers
          fontFamily: 'Calibri'
        });
      } else if (isPackoutItem) {
        applyTextStyle(cell, {
          italic: true,
          fontSize: 10,
          foregroundColor: '#333333',  // Dark gray for packout items
          fontFamily: 'Calibri'
        });
      } else {
        applyTextStyle(cell, {
          bold: cellIndex === 0,  // Only make the first column bold for regular items
          fontSize: 10,
          foregroundColor: '#000000',
          fontFamily: 'Calibri'
        });
      }
      
      // Alternating row colors
      if (rowIndex % 2 === 1) {
        cell.setBackgroundColor('#F0F0F0');
      }
      
      // Cell padding and alignment
      cell.setPaddingTop(0)
        .setPaddingBottom(0)
        .setVerticalAlignment(DocumentApp.VerticalAlignment.CENTER);
    });
  });

  // Table styling
  table.setAttributes({
    [DocumentApp.Attribute.BORDER_WIDTH]: 0
  });

  table.setBorderWidth(0);

  // Set column widths based on the table type
  if (headers.length === 4) {  // Paint table
    setColumnWidths(table, [30, 50, 130, 100, null]);  // [Line, Qty, Product, Sheen, Color]
  } else if (headers.length === 2 && title === 'Rental Equipment') {  // Rental table
    setColumnWidths(table, [30, 200, 100]);  // [Line, Item, Duration]
  } else if (headers.length === 2) {  // Sundries or Equipment table
    setColumnWidths(table, [30, 50, 450]);  // [Line, Qty, Item]
  }

  // Set compact row heights
  setCompactRowHeights(table);

  // Add subtle borders
  addTableBorders(table);
}

function setColumnWidths(table, widths) {
  for (var i = 0; i < table.getNumRows(); i++) {
    var row = table.getRow(i);
    for (var j = 0; j < widths.length && j < row.getNumCells(); j++) {
      if (widths[j] !== null) {
        row.getCell(j).setWidth(widths[j]);
      }
    }
  }
}


function setCompactRowHeights(table) {
  for (var i = 0; i < table.getNumRows(); i++) {
    var row = table.getRow(i);
    row.setMinimumHeight(1);  // Set a compact height for each row
  }
}

function addTableBorders(table) {
  var style = {};
  style[DocumentApp.Attribute.BORDER_WIDTH] = 0.5;  // Thinner borders
  style[DocumentApp.Attribute.BORDER_COLOR] = '#D0D0D0';  // Lighter gray color

  for (var i = 0; i < table.getNumRows(); i++) {
    var row = table.getRow(i);
    for (var j = 0; j < row.getNumCells(); j++) {
      var cell = row.getCell(j);
      cell.setAttributes(style);
    }
  }
}

function insertAdditionalItems(body, sectionTitle, items) {

function parsePaintData(paintString) {
  return paintString.split('\n').filter(Boolean).map(function(item) {
    var match = item.match(/(\d+(?:\.\d+)?)\s*gals\s*(.*?)\s*\((.*?),\s*(.*?)\)/);
    return match ? [match[1] + ' gals', match[2], match[3], match[4]] : ['', item, '', ''];
  });
}

function parseSundriesData(sundriesString) {
  const rows = [];
  if (!sundriesString) return rows;
  
  const lines = sundriesString.split('\n').filter(Boolean);
  
  // First pass - identify packouts and items
  let packoutName = null;
  let packoutItems = [];
  let regularItems = [];
  let packoutQty = "1"; // Default quantity
  
  lines.forEach(function(line) {
    const match = line.match(/(\d+)x\s*(.*)/);
    
    if (match) {
      const qty = match[1];
      const itemText = match[2];
      
      // If it's a packout header
      if (itemText.includes('PACKOUT:')) {
        packoutName = itemText.split('PACKOUT:')[1].trim();
        packoutQty = qty;
      }
      // If it's a packout item (has "Part of")
      else if (itemText.includes('(Part of ')) {
        // If we don't have a packout name yet, extract it from item
        if (!packoutName) {
          const parts = itemText.match(/\(Part of (.*?)\)/);
          if (parts && parts[1]) {
            packoutName = parts[1].trim();
          }
        }
        
        // Add to packout items array
        const baseItem = itemText.split('(Part of ')[0].trim();
        packoutItems.push([qty, baseItem]);
      }
      // Regular item
      else {
        regularItems.push([qty, itemText]);
      }
    }
  });
  
  // Build the final ordered list
  
  // 1. First add the packout header if we found packout items
  if (packoutName && packoutItems.length > 0) {
    // Add packout header
    rows.push([packoutQty, `PACKOUT: ${packoutName}`]);
    
    // Then add packout items with indentation
    packoutItems.forEach(function(item) {
      rows.push([item[0], `---- ${item[1]}`]);
    });
  }
  
  // 2. Then add all regular items
  regularItems.forEach(function(item) {
    rows.push(item);
  });
  
  return rows;
}

function applyTextStyle(element, styles) {
  var textElement = element.editAsText ? element.editAsText() : element;
  if (styles.bold !== undefined) textElement.setBold(styles.bold);
  if (styles.italic !== undefined) textElement.setItalic(styles.italic);
  if (styles.fontSize !== undefined) textElement.setFontSize(styles.fontSize);
  if (styles.fontFamily !== undefined) textElement.setFontFamily(styles.fontFamily);
  if (styles.foregroundColor !== undefined) textElement.setForegroundColor(styles.foregroundColor);
}

function submitRentalForm(data) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    if (!sheet) {
      sheet = ss.insertSheet("RentalSubmissions");
      sheet.appendRow(["Job", "Job Name", "Name", "Notes", "Date", "Rental Items", "Rental Rep", "Rental Rep Email"]);
    }
 
    var inputDate = new Date(data.date);
    var userTimezoneOffset = inputDate.getTimezoneOffset() * 60000;
    var adjustedDate = new Date(inputDate.getTime() + userTimezoneOffset);
    var formattedDate = formatDate(adjustedDate);
 
    var jobCode = data.job;
    var jobName = data.jobName || '';
 
    sheet.appendRow([jobCode, jobName, data.name, data.notes, formattedDate, data.rentalItems,

    var templateId    = getSetting('Rental Order Template');
    var senderName    = getSetting('Sender Name');
    var saveToDrive   = getSetting('Save PDF To Drive').toLowerCase() === 'yes';
    var pdfFolderName = getSetting('PDF Folder Name');
 
    var templateFile = DriveApp.getFileById(templateId);
    var copy = templateFile.makeCopy(jobCode + ' - ' + jobName + ' - Rental Order');
    var doc = DocumentApp.openById(copy.getId());
    var body = doc.getBody();
 
    body.replaceText('{{job}}',      jobCode);
    body.replaceText('{{job_name}}', jobName);
    body.replaceText('{{name}}',     data.name);
    body.replaceText('{{notes}}',    data.notes);
    body.replaceText('{{date}}',     formattedDate);
    body.replaceText('{{vendor}}',   data.rentalRep);
 
    setDocumentStyle(doc);
 
    var itemsPlaceholder = body.findText('{{items}}');
    if (itemsPlaceholder) itemsPlaceholder.getElement().getParent().removeFromParent();
 
    insertItemsTable(body, 'Rental Equipment', parseRentalData(data.rentalItems), ['Qty', 'Item', 'Duration']);
 
    var now = new Date();
    var stampParagraph = body.appendParagraph('Form submitted on: ' + now.toLocaleDateString() + ' at ' + now.toLocaleTimeString());
    applyTextStyle(stampParagraph, { italic: true, fontSize: 9, foregroundColor: '#666666', fontFamily: 'Arial' });
    stampParagraph.setAlignment(DocumentApp.HorizontalAlignment.RIGHT);
 
    doc.saveAndClose();
 
    var pdf = DriveApp.getFileById(copy.getId()).getAs('application/pdf');
 
    var ccRecipients = [];
    if (data.pmEmail    && data.pmEmail.trim()    !== '') ccRecipients.push(data.pmEmail);
    if (data.superEmail && data.superEmail.trim() !== '') ccRecipients.push(data.superEmail);
    if (data.foreman    && data.foreman.trim()    !== '') ccRecipients.push(data.foreman);
    var ccString = ccRecipients.join(',');
 
    var emailBody = buildRentalEmail(data, jobCode, jobName, formattedDate);
 
    GmailApp.sendEmail(data.rentalRepEmail, jobCode + ' - ' + jobName + ' - Rental Order', '', {
      cc: ccString, attachments: [pdf], name: senderName, htmlBody: emailBody
    });
 
    if (saveToDrive) {
      var folders = DriveApp.getFoldersByName(pdfFolderName);
      var folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(pdfFolderName);
      DriveApp.getFileById(copy.getId()).moveTo(folder);
    } else {
      DriveApp.getFileById(copy.getId()).setTrashed(true);
    }
 
    return 'Rental order submitted successfully!';
  } catch (error) {
    console.error('Error in submitRentalForm:', error);
    return 'An error occurred: ' + error.toString();
  }

// Update order status
function updateOrderStatus(orderType, orderId, newStatus) {
  var sheetName;
  switch(orderType) {
    case 'equipment':
      sheetName = "EquipmentSubmissions";
      break;
    case 'wallcovering':
      sheetName = "WallcoveringSubmissions";
      break;
    case 'hauloff':
      sheetName = "HaulOffSubmissions";
      break;
    default:
      throw new Error("Invalid order type");
  }
  
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (!sheet) throw new Error("Sheet not found");
  
  // Ensure status column exists (column J, index 9)
  var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  if (headers.length < 10 || headers[9] !== 'Status') {
    sheet.getRange(1, 10).setValue('Status');
  }
  
  // Update the status
  sheet.getRange(orderId + 1, 10).setValue(newStatus);
  
  return "Status updated successfully";
}

// Helper function to ensure a sheet exists with proper headers
function ensureSheetExists(sheetName, headers) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(sheetName);
  
  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
  } else {
    // Ensure the sheet has all required headers
    var currentHeaders = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    if (currentHeaders.length < headers.length) {
      sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
    }
  }
}

// Export orders to PDF
function exportOrdersToPDF(orderType) {
  var sheetName;
  switch(orderType) {
    case 'equipment':
      sheetName = "EquipmentSubmissions";
      break;
    case 'wallcovering':
      sheetName = "WallcoveringSubmissions";
      break;
    case 'hauloff':
      sheetName = "HaulOffSubmissions";
      break;
    default:
      throw new Error("Invalid order type");
  }
  
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(sheetName);
  
  if (!sheet) throw new Error("Sheet not found");
  
  // Create a temporary spreadsheet for the export
  var tempSS = SpreadsheetApp.create("Temp Export - " + sheetName);
  var tempSheet = tempSS.getActiveSheet();
  
  // Copy the data
  var range = sheet.getDataRange();
  range.copyTo(tempSheet.getRange(1, 1));
  
  // Format the export
  tempSheet.setFrozenRows(1);
  tempSheet.autoResizeColumns(1, tempSheet.getLastColumn());
  
  // Convert to PDF
  var blob = tempSS.getAs('application/pdf');
  blob.setName(sheetName + '_Export_' + new Date().toISOString().split('T')[0] + '.pdf');
  
  // Delete temporary spreadsheet
  DriveApp.getFileById(tempSS.getId()).setTrashed(true);
  
  return blob;
}

// Search across all order types
function searchOrders(searchTerm) {
  searchTerm = searchTerm.toLowerCase();
  var results = {
    equipment: [],
    wallcovering: [],
    hauloff: []
  };
  
  // Search equipment orders
  var equipmentOrders = getEquipmentOrders();
  results.equipment = equipmentOrders.filter(order => {
    return order.job.toLowerCase().includes(searchTerm) ||
           order.name.toLowerCase().includes(searchTerm) ||
           order.pm.toLowerCase().includes(searchTerm) ||
           order.super.toLowerCase().includes(searchTerm);
  });
  
  // Search wallcovering orders
  var wallcoveringOrders = getWallcoveringOrders();
  results.wallcovering = wallcoveringOrders.filter(order => {
    return order.job.toLowerCase().includes(searchTerm) ||
           order.name.toLowerCase().includes(searchTerm) ||
           order.pm.toLowerCase().includes(searchTerm) ||
           order.super.toLowerCase().includes(searchTerm);
  });
  
  // Search haul off orders
  var hauloffOrders = getHauloffOrders();
  results.hauloff = hauloffOrders.filter(order => {
    return order.job.toLowerCase().includes(searchTerm) ||
           order.name.toLowerCase().includes(searchTerm) ||
           order.pm.toLowerCase().includes(searchTerm) ||
           order.super.toLowerCase().includes(searchTerm);
  });
  
  return results;
}

// Get dashboard statistics
function getDashboardStats() {
  var stats = {
    total: {
      equipment: 0,
      wallcovering: 0,
      hauloff: 0
    },
    byStatus: {
      new: { equipment: 0, wallcovering: 0, hauloff: 0 },
      pending: { equipment: 0, wallcovering: 0, hauloff: 0 },
      completed: { equipment: 0, wallcovering: 0, hauloff: 0 }
    },
    recentOrders: []
  };
  
  // Get counts from each sheet
  var equipmentOrders = getEquipmentOrders();
  var wallcoveringOrders = getWallcoveringOrders();
  var hauloffOrders = getHauloffOrders();
  
  // Calculate totals
  stats.total.equipment = equipmentOrders.length;
  stats.total.wallcovering = wallcoveringOrders.length;
  stats.total.hauloff = hauloffOrders.length;
  
  // Calculate by status
  equipmentOrders.forEach(order => {
    var status = (order.status || 'new').toLowerCase();
    if (stats.byStatus[status]) {
      stats.byStatus[status].equipment++;
    }
  });
  
  wallcoveringOrders.forEach(order => {
    var status = (order.status || 'new').toLowerCase();
    if (stats.byStatus[status]) {
      stats.byStatus[status].wallcovering++;
    }
  });
  
  hauloffOrders.forEach(order => {
    var status = (order.status || 'new').toLowerCase();
    if (stats.byStatus[status]) {
      stats.byStatus[status].hauloff++;
    }
  });
  
  // Get recent orders (last 5 from each type)
  var allOrders = [];
  
  equipmentOrders.slice(-5).forEach(order => {
    order.type = 'equipment';
    allOrders.push(order);
  });
  
  wallcoveringOrders.slice(-5).forEach(order => {
    order.type = 'wallcovering';
    allOrders.push(order);
  });
  
  hauloffOrders.slice(-5).forEach(order => {
    order.type = 'hauloff';
    allOrders.push(order);
  });
  
  // Sort by date and get the 10 most recent
  stats.recentOrders = allOrders
    .sort((a, b) => new Date(b.date) - new Date(a.date))
    .slice(0, 10);
  
  return stats;
}

function detectJobsSheetLayout(jobsSheet) {
  var headerB = jobsSheet.getRange('B1').getValue().toString().toLowerCase().trim();
  var headerC = jobsSheet.getRange('C1').getValue().toString().toLowerCase().trim();
  if (headerB.indexOf('job name') !== -1) return 'five';
  if (headerB === 'pm' && headerC.indexOf('super') !== -1) return 'five';

  var lastRow = jobsSheet.getLastRow();
  if (lastRow < 2) return 'empty';

  var row = jobsSheet.getRange(2, 1, 1, 5).getValues()[0];
  if (row[4] && row[4].toString().trim()) return 'five';

  var b = row[1] ? row[1].toString().trim() : '';
  var d = row[3] ? row[3].toString().trim() : '';
  var dLooksLikeAddress = d && (/\d/.test(d) || /st|ave|rd|blvd|ste|suite|street|clay|way|dr/i.test(d));
  if (b && dLooksLikeAddress) return 'four';

  return 'unknown';
}

/**
 * Migrates Jobs sheet from 4 columns (Job #, PM, Super, Address)
 * to 5 columns (Job Code, Job Name, PM, Super, Address).
 */
function migrateJobsToFiveColumns() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var ui = SpreadsheetApp.getUi();
  var jobsSheet = ss.getSheetByName('Jobs');

  if (!jobsSheet) {
    ui.alert('Error', 'Jobs sheet not found.', ui.ButtonSet.OK);
    return;
  }

  var layout = detectJobsSheetLayout(jobsSheet);
  if (layout === 'five') {
    ui.alert('Already Migrated', 'Jobs sheet already uses the 5-column layout (Job Code, Job Name, PM, Super, Address). No migration needed.', ui.ButtonSet.OK);
    return;
  }

  if (layout === 'empty') {
    jobsSheet.getRange(1, 1, 1, 5).setValues([['Job Code', 'Job Name', 'PM', 'Super', 'Address']]);
    jobsSheet.getRange(1, 1, 1, 5).setFontWeight('bold');
    ui.alert('Headers Added', 'Row 1 headers were set. Add job rows starting in row 2.', ui.ButtonSet.OK);
    return;
  }

  var confirm = ui.alert(
    'Migrate Jobs Sheet',
    'This will convert your Jobs sheet to:\n\n' +
    'A = Job Code (e.g. 1097)\n' +
    'B = Job Name (e.g. Citi Bank)\n' +
    'C = PM\n' +
    'D = Super\n' +
    'E = Address\n\n' +
    'A backup copy of the sheet will be created first. Continue?',
    ui.ButtonSet.YES_NO
  );
  if (confirm !== ui.Button.YES) return;

  var lastRow = jobsSheet.getLastRow();
  var oldData = jobsSheet.getRange(2, 1, lastRow - 1, 4).getValues();

  var backupName = 'Jobs Backup ' + Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HHmm');
  ss.getSheets().forEach(function(sheet) {
    if (sheet.getName() === backupName) {
      ss.deleteSheet(sheet);
    }
  });
  jobsSheet.copyTo(ss).setName(backupName);

  var newData = oldData.map(function(row) {
    var split = splitJobCodeAndName(row[0]);
    return [
      split.code,
      split.name,
      row[1] ? row[1].toString().trim() : '',
      row[2] ? row[2].toString().trim() : '',
      row[3] ? row[3].toString().trim() : ''
    ];
  }).filter(function(row) {
    return row[0] || row[1] || row[2] || row[3] || row[4];
  });

  jobsSheet.getRange(1, 1, 1, 5).setValues([['Job Code', 'Job Name', 'PM', 'Super', 'Address']]);
  jobsSheet.getRange(1, 1, 1, 5).setFontWeight('bold').setBackground('#f3f4f6');

  if (newData.length > 0) {
    jobsSheet.getRange(2, 1, newData.length, 5).setValues(newData);
  }

  jobsSheet.setColumnWidth(1, 100);
  jobsSheet.setColumnWidth(2, 200);
  jobsSheet.setColumnWidth(3, 140);
  jobsSheet.setColumnWidth(4, 140);
  jobsSheet.setColumnWidth(5, 280);

  ui.alert(
    'Migration Complete',
    'Migrated ' + newData.length + ' job rows.\n\n' +
    'Backup: "' + backupName + '"\n\n' +
    'Example: "1097 Citi Bank" → Code: 1097, Name: Citi Bank\n\n' +
    'Reload the Field Request form and pick a job to verify PM/Super emails.',
    ui.ButtonSet.OK
  );
}

    var jobsLastRow = jobsSheet.getLastRow();
    if (jobsLastRow > 1) {
      var jobsRange = jobsSheet.getRange("A2:E" + jobsLastRow);
      var jobsValues = jobsRange.getValues();
      
      var cleanedJobs = jobsValues.map(function(row) {
        var originalCode = row[0];
        var originalName = row[1];
        var originalPM = row[2];
        var originalSuper = row[3];
        var originalAddress = row[4];
        
        var cleanCode = row[0];
        if (cleanCode) {
          cleanCode = cleanCode.toString()
            .trim()
            .replace(/'/g, '')
            .replace(/'/g, '')
            .replace(/`/g, '');
        }
        
        var cleanName = row[1] ? row[1].toString().trim() : '';
        var cleanPM = row[2] ? row[2].toString().trim() : '';
        var cleanSuper = row[3] ? row[3].toString().trim() : '';
        var cleanAddress = row[4] ? row[4].toString().trim() : '';
        
        if (originalCode !== cleanCode || originalName !== cleanName ||
            originalPM !== cleanPM || originalSuper !== cleanSuper ||
            originalAddress !== cleanAddress) {
          cleanupCount++;
        }
        
        return [cleanCode || '', cleanName || '', cleanPM || '', cleanSuper || '', cleanAddress || ''];
      });
      
      jobsRange.setValues(cleanedJobs);
    }
  }
  
  // Clean PMs sheet - just the names
  var pmSheet = ss.getSheetByName("PMs");
  if (pmSheet) {
    var pmLastRow = pmSheet.getLastRow();
    if (pmLastRow > 1) {
      var pmRange = pmSheet.getRange("A2:A" + pmLastRow);
      var pmValues = pmRange.getValues();
      
      var cleanedPMs = pmValues.map(function(row) {
        var originalName = row[0];
        var cleanName = row[0];
        
        if (cleanName) {
          cleanName = cleanName.toString().trim();
        }
        
        if (originalName !== cleanName) {
          cleanupCount++;
        }
        
        return [cleanName || ''];
      });
      
      pmRange.setValues(cleanedPMs);
    }
  }
  
  // Clean Supers sheet - just the names
  var superSheet = ss.getSheetByName("Supers");
  if (superSheet) {
    var superLastRow = superSheet.getLastRow();
    if (superLastRow > 1) {
      var superRange = superSheet.getRange("A2:A" + superLastRow);
      var superValues = superRange.getValues();
      
      var cleanedSupers = superValues.map(function(row) {
        var originalName = row[0];
        var cleanName = row[0];
        
        if (cleanName) {
          cleanName = cleanName.toString().trim();
        }
        
        if (originalName !== cleanName) {
          cleanupCount++;
        }
        
        return [cleanName || ''];
      });
      
      superRange.setValues(cleanedSupers);
    }
  }
  
  // Show summary
  ui.alert('Cleanup Complete!', 
           'Cleaned ' + cleanupCount + ' entries:\n\n' +
           '• Trimmed Job Code, Job Name, PM, Super, and Address\n' +
           '• Removed apostrophes from Job Code\n' +
           '• Trimmed PM and Super names on PMs/Supers sheets', 
           ui.ButtonSet.OK);
}

  }

// Save brushout data from the spreadsheet BrushoutForm dialog (merge mode).
function saveBrushout(data) {
  var colors = (data.colors || []).map(function(c) {
    return c ? c.toString().trim() : '';
  }).filter(function(c) { return c; });

  if (!colors.length) {
    throw new Error('No colors to save.');
  }

  var result = handleBrushoutsData({
    job_number: data.jobNumber,
    paint_vendor: data.paintVendor,
    paint_items: colors,
    mode: 'merge'
  });
  var parsed = JSON.parse(result.getContent());
  if (!parsed.success) {
    throw new Error(parsed.error || 'Failed to save brushout.');
  }
  return 'Success';
}

function getBrushoutSheet_(ss) {
  var sheet = ss.getSheetByName('BrushOuts');
  if (!sheet) {
    sheet = ss.getSheetByName('Brush Outs');
  }
  if (!sheet) {
    sheet = ss.insertSheet('BrushOuts');
    writeBrushoutHeaders_(sheet, BRUSHOUT_MAX_COLORS, false);
  } else {
    ensureBrushoutColorHeaders_(sheet, BRUSHOUT_MAX_COLORS);
  }
  return sheet;
}

function buildColorHeaderLabels_(count) {
  var headers = [];
  for (var i = 1; i <= count; i++) {
    headers.push('Color ' + i);
  }
  return headers;
}

/** Field Request Order default: Job Number | Paint | Color 1… */
function writeBrushoutHeaders_(sheet, colorCount, withApproved) {
  var colorHeaders = buildColorHeaderLabels_(colorCount);
  var headers;
  if (withApproved) {
    headers = ['Job Number and Name', 'Approved', 'Paint'].concat(colorHeaders);
  } else {
    headers = ['Job Number', 'Paint'].concat(colorHeaders);
  }
  writeSheetRow_(sheet, 1, headers);
  sheet.getRange(1, 1, 1, headers.length).setFontWeight('bold');
  sheet.setFrozenRows(1);
}

function detectBrushoutLayout_(headerRow) {
  var h2 = (headerRow[1] || '').toString().toLowerCase();

  if (h2 === 'approved' || h2.indexOf('approv') >= 0) {
    return { paintCol: 3, colorStartCol: 4, hasApproved: true, jobCol: 1 };
  }
  if (h2 === 'paint' || h2.indexOf('vendor') >= 0 || h2.indexOf('paint') >= 0) {
    return { paintCol: 2, colorStartCol: 3, hasApproved: false, jobCol: 1 };
  }
  var h3 = (headerRow[2] || '').toString().toLowerCase();
  if (h3 === 'paint' || h3.indexOf('paint') >= 0) {
    return { paintCol: 3, colorStartCol: 4, hasApproved: true, jobCol: 1 };
  }
  return { paintCol: 2, colorStartCol: 3, hasApproved: false, jobCol: 1 };
}

function ensureBrushoutColorHeaders_(sheet, minColors) {
  var lastCol = Math.max(sheet.getLastColumn(), 2);
  var headerRow = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
  var layout = detectBrushoutLayout_(headerRow);
  var existingColorCols = Math.max(0, lastCol - layout.colorStartCol + 1);

  if (existingColorCols < minColors) {
    var newHeaders = [];
    for (var i = existingColorCols + 1; i <= minColors; i++) {
      newHeaders.push('Color ' + i);
    }
    if (newHeaders.length) {
      var startCol = layout.colorStartCol + existingColorCols;
      writeSheetRow_(sheet, 1, newHeaders, startCol);
      sheet.getRange(1, startCol, 1, startCol + newHeaders.length - 1).setFontWeight('bold');
    }
  }
  return layout;
}

/** Write one row — always matches range width to data length. */
function writeSheetRow_(sheet, row, values, startCol) {
  startCol = startCol || 1;
  if (!values || !values.length) return;
  sheet.getRange(row, startCol, 1, values.length).setValues([values]);
}

/**
 * Merge key from "Label - FL - Color" (label + floor; color changes = same slot).
 */
function brushoutMergeKey_(line) {
  var s = (line || '').toString().trim();
  if (!s) return '';
  var parts = s.split(' - ').map(function(p) { return p.trim(); }).filter(function(p) { return p; });
  if (parts.length >= 3) {
    return (parts[0] + '|' + parts[1]).toLowerCase();
  }
  if (parts.length === 2) {
    return parts[0].toLowerCase();
  }
  return s.toLowerCase();
}

function readColorCells_(rowValues, colorStartCol, maxColors) {
  var colors = [];
  for (var c = 0; c < maxColors; c++) {
    var val = rowValues[colorStartCol - 1 + c];
    colors.push(val ? val.toString().trim() : '');
  }
  while (colors.length && !colors[colors.length - 1]) {
    colors.pop();
  }
  return colors;
}

function mergeBrushoutColors_(existingColors, incomingItems, maxColors) {
  var colorCols = existingColors.slice();
  var keyToIndex = {};

  for (var i = 0; i < colorCols.length; i++) {
    var existingKey = brushoutMergeKey_(colorCols[i]);
    if (existingKey) {
      keyToIndex[existingKey] = i;
    }
  }

  for (var j = 0; j < incomingItems.length; j++) {
    var line = (incomingItems[j] || '').toString().trim();
    if (!line) continue;
    var key = brushoutMergeKey_(line);
    if (!key) continue;

    if (Object.prototype.hasOwnProperty.call(keyToIndex, key)) {
      colorCols[keyToIndex[key]] = line;
    } else {
      var placed = false;
      for (var k = 0; k < colorCols.length; k++) {
        if (!colorCols[k]) {
          colorCols[k] = line;
          keyToIndex[key] = k;
          placed = true;
          break;
        }
      }
      if (!placed) {
        if (colorCols.length >= maxColors) {
          throw new Error('BrushOuts limit reached (' + maxColors + ' colors per job).');
        }
        keyToIndex[key] = colorCols.length;
        colorCols.push(line);
      }
    }
  }

  return colorCols;
}

function buildBrushoutRowData_(layout, jobKey, vendor, mergedColors) {
  var totalCols = layout.colorStartCol - 1 + mergedColors.length;
  var rowData = [];
  for (var c = 0; c < totalCols; c++) {
    rowData.push('');
  }
  rowData[layout.jobCol - 1] = jobKey;
  rowData[layout.paintCol - 1] = vendor;
  for (var ci = 0; ci < mergedColors.length; ci++) {
    rowData[layout.colorStartCol - 1 + ci] = mergedColors[ci];
  }
  return rowData;
}

function jsonResponse_(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

/**
 * Handle BrushOuts data from JobFlow / desktop.
 *
 * Body:
 *   job_number   — full "123 Job Name" string
 *   paint_vendor — PPG, etc.
 *   paint_items  — ["Label - FL - SW7004", ...]
 *   mode         — "merge" (default) or "replace"
 */
function handleBrushoutsData(data) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var brushoutSheet = getBrushoutSheet_(ss);
    var layout = ensureBrushoutColorHeaders_(brushoutSheet, BRUSHOUT_MAX_COLORS);

    if (!data.job_number || !data.paint_vendor || !data.paint_items || !Array.isArray(data.paint_items)) {
      return jsonResponse_({
        success: false,
        error: 'Missing required data: job_number, paint_vendor, and paint_items array'
      });
    }

    var mode = (data.mode || 'merge').toString().toLowerCase();
    if (mode !== 'merge' && mode !== 'replace') {
      mode = 'merge';
    }

    var incoming = data.paint_items
      .map(function(item) { return (item || '').toString().trim(); })
      .filter(function(item) { return item; });

    if (!incoming.length) {
      return jsonResponse_({
        success: false,
        error: 'No paint_items to save.'
      });
    }

    if (mode === 'replace' && incoming.length > BRUSHOUT_MAX_COLORS) {
      incoming = incoming.slice(0, BRUSHOUT_MAX_COLORS);
    }

    var existingData = brushoutSheet.getDataRange().getValues();
    var rowToUpdate = -1;
    var jobKey = data.job_number.toString().trim();

    for (var i = 1; i < existingData.length; i++) {
      if (existingData[i][layout.jobCol - 1].toString().trim() === jobKey) {
        rowToUpdate = i + 1;
        break;
      }
    }

    var mergedColors;
    if (mode === 'replace' || rowToUpdate <= 0) {
      mergedColors = incoming.slice(0, BRUSHOUT_MAX_COLORS);
    } else {
      var existingRow = existingData[rowToUpdate - 1];
      var existingColors = readColorCells_(existingRow, layout.colorStartCol, BRUSHOUT_MAX_COLORS);
      mergedColors = mergeBrushoutColors_(existingColors, incoming, BRUSHOUT_MAX_COLORS);
    }

    if (mergedColors.length > BRUSHOUT_MAX_COLORS) {
      mergedColors = mergedColors.slice(0, BRUSHOUT_MAX_COLORS);
    }

    var rowData = buildBrushoutRowData_(layout, jobKey, data.paint_vendor, mergedColors);
    var targetRow;

    if (rowToUpdate > 0) {
      writeSheetRow_(brushoutSheet, rowToUpdate, rowData);
      targetRow = rowToUpdate;
      Logger.log('Merged brushout row for job: ' + jobKey + ' (' + incoming.length + ' incoming, ' + mergedColors.length + ' total)');
    } else {
      targetRow = brushoutSheet.getLastRow() + 1;
      writeSheetRow_(brushoutSheet, targetRow, rowData);
      Logger.log('Added brushout row for job: ' + jobKey);
    }

    if (layout.hasApproved) {
      brushoutSheet.getRange(targetRow, 2).insertCheckboxes();
    }

    return jsonResponse_({
      success: true,
      message: mode === 'merge'
        ? 'BrushOuts merged successfully (' + incoming.length + ' line(s), ' + mergedColors.length + ' on sheet).'
        : 'BrushOuts saved successfully.',
      job_number: jobKey,
      paint_vendor: data.paint_vendor,
      mode: mode,
      colors_added: incoming.length,
      colors_on_sheet: mergedColors.length,
      row_updated: targetRow
    });

  } catch (error) {
    Logger.log('Error in handleBrushoutsData: ' + error.toString());
    return jsonResponse_({
      success: false,
      error: 'Failed to save BrushOuts data: ' + error.toString()
    });
  }
}

/** Field order form — read all color columns for a job. */
function getBrushoutColors(jobNumber) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var brushoutSheet = ss.getSheetByName('BrushOuts') || ss.getSheetByName('Brush Outs');
    if (!brushoutSheet) {
      return { colors: [], vendor: '' };
    }

    var data = brushoutSheet.getDataRange().getValues();
    if (data.length < 2) {
      return { colors: [], vendor: '' };
    }

    var layout = detectBrushoutLayout_(data[0]);
    var jobKey = jobNumber.toString().trim();

    for (var i = 1; i < data.length; i++) {
      var row = data[i];
      if ((row[layout.jobCol - 1] || '').toString().trim() !== jobKey) {
        continue;
      }
      return {
        colors: readColorCells_(row, layout.colorStartCol, BRUSHOUT_MAX_COLORS),
        vendor: (row[layout.paintCol - 1] || '').toString().trim()
      };
    }

    return { colors: [], vendor: '' };
  } catch (error) {
    Logger.log('Error in getBrushoutColors: ' + error.toString());
    return { colors: [], vendor: '' };
  }
}
